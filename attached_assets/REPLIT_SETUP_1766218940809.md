# Replit Integration Setup Guide

Complete guide for setting up the 3-app Replit integration with 8825.

## Overview

Three Replit apps that share infrastructure:

1. **8825 Portal** - Web UI for submitting export jobs
2. **8825 Worker** - Scheduled job processor
3. **8825 Sandbox** - Forkable template for contributors

---

## Prerequisites

- [ ] 8825 API Gateway running locally (port 8000)
- [ ] Export Appliance running locally (port 8080)
- [ ] Tailscale or ngrok installed
- [ ] Replit account (Teams recommended for RBAC)
- [ ] Agent token created

---

### Create Database Schema

Create `lib/db.ts`:

```typescript
import Database from "@replit/database";

const db = new Database();

export interface ExportJob {
  id: string;
  user_id: string;
  markdown: string;
  target: "email" | "sms" | "docx" | "pdf" | "gdoc";
  template_id: string;
  status: "pending" | "processing" | "complete" | "failed";
  artifact_url?: string;
  error?: string;
  created_at: string;
  completed_at?: string;
}

export async function createJob(job: Omit<ExportJob, "id" | "created_at">): Promise<string> {
  const id = `job_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  const created_at = new Date().toISOString();
  
  await db.set(`job:${id}`, {
    ...job,
    id,
    created_at,
  });
  
  return id;
}

export async function getJob(id: string): Promise<ExportJob | null> {
  const job = await db.get(`job:${id}`);
  return job as ExportJob | null;
}

export async function updateJob(id: string, updates: Partial<ExportJob>): Promise<void> {
  const job = await getJob(id);
  if (!job) throw new Error("Job not found");
  
  await db.set(`job:${id}`, {
    ...job,
    ...updates,
  });
}

export async function listJobs(limit: number = 20): Promise<ExportJob[]> {
  const keys = await db.list("job:");
  const jobs = await Promise.all(
    keys.slice(0, limit).map(key => db.get(key))
  );
  
  return jobs
    .filter(Boolean)
    .sort((a, b) => 
      new Date(b.created_at).getTime() - new Date(a.created_at).getTime()
    ) as ExportJob[];
}
```

### Create API Route

Create `app/api/export/route.ts`:

```typescript
import { NextRequest, NextResponse } from "next/server";
import { createJob } from "@/lib/db";

export async function POST(request: NextRequest) {
  try {
    const body = await request.json();
    const { markdown, target, template_id = "generic" } = body;
    
    if (!markdown || !target) {
      return NextResponse.json(
        { error: "Missing required fields" },
        { status: 400 }
      );
    }
    
    const jobId = await createJob({
      user_id: "anonymous", // TODO: Add auth
      markdown,
      target,
      template_id,
      status: "pending",
    });
    
    return NextResponse.json({
      job_id: jobId,
      status: "pending",
      message: "Job created successfully",
    });
  } catch (error) {
    console.error("Error creating job:", error);
    return NextResponse.json(
      { error: "Internal server error" },
      { status: 500 }
    );
  }
}
```

### Create Job Status Route

Create `app/api/jobs/[id]/route.ts`:

```typescript
import { NextRequest, NextResponse } from "next/server";
import { getJob } from "@/lib/db";

export async function GET(
  request: NextRequest,
  { params }: { params: { id: string } }
) {
  try {
    const job = await getJob(params.id);
    
    if (!job) {
      return NextResponse.json(
        { error: "Job not found" },
        { status: 404 }
      );
    }
    
    return NextResponse.json(job);
  } catch (error) {
    console.error("Error fetching job:", error);
    return NextResponse.json(
      { error: "Internal server error" },
      { status: 500 }
    );
  }
}
```

### Create Main Page

Replace `app/page.tsx`:

```typescript
"use client";

import { useState } from "react";

export default function Home() {
  const [markdown, setMarkdown] = useState("");
  const [target, setTarget] = useState("email");
  const [jobId, setJobId] = useState<string | null>(null);
  const [status, setStatus] = useState<string | null>(null);
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    setLoading(true);
    setJobId(null);
    setStatus(null);

    try {
      const response = await fetch("/api/export", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ markdown, target }),
      });

      const data = await response.json();

      if (response.ok) {
        setJobId(data.job_id);
        setStatus("pending");
        pollJobStatus(data.job_id);
      } else {
        setStatus(`Error: ${data.error}`);
      }
    } catch (error) {
      setStatus(`Error: ${error.message}`);
    } finally {
      setLoading(false);
    }
  };

  const pollJobStatus = async (id: string) => {
    const interval = setInterval(async () => {
      try {
        const response = await fetch(`/api/jobs/${id}`);
        const job = await response.json();

        setStatus(job.status);

        if (job.status === "complete" || job.status === "failed") {
          clearInterval(interval);
        }
      } catch (error) {
        console.error("Error polling job:", error);
      }
    }, 2000);
  };

  return (
    <main className="min-h-screen p-8 bg-gray-50">
      <div className="max-w-4xl mx-auto">
        <h1 className="text-4xl font-bold mb-8">8825 Export Portal</h1>

        <form onSubmit={handleSubmit} className="space-y-6 bg-white p-6 rounded-lg shadow">
          <div>
            <label className="block text-sm font-medium mb-2">
              Markdown Content
            </label>
            <textarea
              value={markdown}
              onChange={(e) => setMarkdown(e.target.value)}
              className="w-full h-64 p-3 border rounded-md font-mono text-sm"
              placeholder="# Your Markdown Here&#10;&#10;Paste your content..."
              required
            />
          </div>

          <div>
            <label className="block text-sm font-medium mb-2">
              Export Target
            </label>
            <select
              value={target}
              onChange={(e) => setTarget(e.target.value)}
              className="w-full p-3 border rounded-md"
            >
              <option value="email">Email Text</option>
              <option value="sms">SMS Text</option>
              <option value="docx">DOCX Document</option>
              <option value="pdf">PDF Document</option>
              <option value="gdoc">Google Doc</option>
            </select>
          </div>

          <button
            type="submit"
            disabled={loading}
            className="w-full bg-blue-600 text-white py-3 rounded-md font-medium hover:bg-blue-700 disabled:bg-gray-400"
          >
            {loading ? "Submitting..." : "Export"}
          </button>
        </form>

        {jobId && (
          <div className="mt-6 p-6 bg-white rounded-lg shadow">
            <h2 className="text-xl font-semibold mb-4">Job Status</h2>
            <p className="mb-2">
              <span className="font-medium">Job ID:</span> {jobId}
            </p>
            <p>
              <span className="font-medium">Status:</span>{" "}
              <span className={`px-2 py-1 rounded ${
                status === "complete" ? "bg-green-100 text-green-800" :
                status === "failed" ? "bg-red-100 text-red-800" :
                status === "processing" ? "bg-yellow-100 text-yellow-800" :
                "bg-gray-100 text-gray-800"
              }`}>
                {status}
              </span>
            </p>
          </div>
        )}
      </div>
    </main>
  );
}
```

### Create Worker Script

Create `worker.py`:

```python
import os
import time
import requests
from replit import db

GATEWAY_URL = os.environ.get("GATEWAY_URL")
AGENT_TOKEN = os.environ.get("AGENT_TOKEN")

def process_jobs():
    """Process pending jobs from Replit DB."""
    print("Checking for pending jobs...")
    
    # Get all job keys
    job_keys = [k for k in db.keys() if k.startswith("job:")]
    
    for key in job_keys:
        job = db[key]
        
        if job.get("status") != "pending":
            continue
        
        print(f"Processing job {job['id']}...")
        
        # Update status to processing
        job["status"] = "processing"
        db[key] = job
        
        try:
            # Call 8825 Gateway
            response = requests.post(
                f"{GATEWAY_URL}/export/render",
                json={
                    "markdown": job["markdown"],
                    "target": job["target"],
                    "template_id": job.get("template_id", "generic"),
                    "mode": "paste"
                },
                headers={"X-8825-Token": AGENT_TOKEN},
                timeout=30
            )
            
            if response.ok:
                result = response.json()
                job["status"] = "complete"
                job["artifact_url"] = result.get("download_url")
                job["completed_at"] = time.strftime("%Y-%m-%dT%H:%M:%SZ")
                print(f"✅ Job {job['id']} completed")
            else:
                job["status"] = "failed"
                job["error"] = f"Gateway error: {response.text}"
                print(f"❌ Job {job['id']} failed: {response.text}")
            
            db[key] = job
            
        except Exception as e:
            print(f"❌ Job {job['id']} failed: {e}")
            job["status"] = "failed"
            job["error"] = str(e)
            db[key] = job

def main():
    """Main worker loop."""
    print("8825 Worker started")
    print(f"Gateway URL: {GATEWAY_URL}")
    
    while True:
        try:
            process_jobs()
        except Exception as e:
            print(f"Error in worker loop: {e}")
        
        # Wait 1 minute before next check
        time.sleep(60)

if __name__ == "__main__":
    main()
```

### Configure Secrets

In Replit Secrets:

```

AGENT_TOKEN=8825_your_agent_token_here
```

### Set Up Scheduled Deployment

1. Click "Deploy" in Replit

### Create Mock Gateway

Create `mock_gateway.py`:

```python
from fastapi import FastAPI, Header, HTTPException
from pydantic import BaseModel
import uvicorn

app = FastAPI(title="8825 Mock Gateway")

class ExportRequest(BaseModel):
    markdown: str
    target: str
    template_id: str = "generic"
    mode: str = "paste"

@app.get("/health")
def health():
    return {"status": "healthy", "mock": True}

@app.post("/export/render")
def mock_render(request: ExportRequest, x_8825_token: str = Header(None)):
    if not x_8825_token:
        raise HTTPException(401, "Missing token")
    
    return {
        "success": True,
        "job_id": "mock_123",
        "clean_text": f"Mock export of: {request.markdown[:50]}...",
        "download_url": "/jobs/mock_123/download"
    }

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

- [ ] Only API Gateway exposed via tunnel

- [ ] Agent token has limited scopes

- [ ] Audit logging active
- [ ] Tokens stored securely

## Troubleshooting

### Portal can't reach Gateway

```bash
# Test tunnel
curl https://your-machine.ts.net/health

# Check gateway logs
# (in terminal running server.py)
```

### Worker not processing jobs

```bash
# Check worker logs in Replit
# Verify GATEWAY_URL and AGENT_TOKEN in Secrets
# Test token manually:
curl -H "X-8825-Token: YOUR_TOKEN" https://your-machine.ts.net/health
```

### Token rejected

```bash
# Verify token exists
python token_cli.py list

# Check audit logs
tail ~/.config/8825/audit_logs/audit_*.jsonl
```
